assessment API

POST: /api/assessment/create                        // For creating an assessment
    params:
        header:
            x-auth-token                            // token for verifying that admin is logged in
        body:
            assessment:
                name
                subheading
                category
                difficulty
                description
                time
                questions
    return:
        status_code
        error: false/true
        message: "..string"

GET: /api/assessment/view/:id            // Get all questions of assessment
    return:
        status_code
        error: false/true
        message: "..string"
        data:   { //whole questions object}
