
------------------- Level API --------------------

GET: /api/level
    params:
        header:
            x-auth-token                            // token for verifying that any user is logged in
        query:
            range
    return:
        status_code
        error: false/true
        message: "..string"
        data: [{
            id                      // Same as name
            name
            subheading
            category
            difficulty
            type
            image_url
            game_url
            description
            qualification_iq
            rules: []
        }, ...]


GET: /api/level/<name>            // Get particular level information
    params:
        header:
            x-auth-token                            // token for verifying that any user is logged in
    return:
        status_code
        error: false/true
        message: "..string"
        data:   {
            name
            subheading
            category
            type
            difficulty
            description
            image_url
            game_url
            rules
            hints
            qualification_iq
            attributes
        }

POST: /api/level/                             // For creating a new level
    params:
        header:
            x-auth-token                            // token for verifying that admin is logged in
        body:
            level:
                name*
                subheading*
                category*
                difficulty*
                type*
                description
                image_url*
                rules
                hints
                qualification_iq*
                Attributes
    return:
        status_code
        error: false/true
        message: {
            '_id': <Level_Name> 
        }

PUT: /api/level/<name>             // Modifying level information
    params:
        header:
            x-auth-token    //token for verifying that admin is logged in
    body:
        //whatever final level you want
    return:
        status_code
        error: false/true
        message: "..string"
        data: {New Level Object}

DELETE: /api/level/<name>		// Delete level with this name
    params:
        header:
            x-auth-token 	//token for verifying that admin is logged in
    return:
        status_code
        error: false/true
        message: "..string"

POST: /api/level/modify_secret   // Gets the top 10 players of that level
    params:
        header:
            x-auth-token
    body:
        name
        old_secret
        new_secret          
    returns:
        status_code
        error: false/true
        message: "..string"
        data: {
            name: level_name,
            leaderboard: []
        }
