            user_id

            university
            security_question_id
            security_answer
            
    return:
        status_code
        error: false/true
        message: "..string"


POST: /api/user/login       // for loggin users in
    params:
        body:
            user_id
            password
    return:
        status_code
        error: false/true
        message: "..string"
        data:   {
            token           // JWT token for logging user in (expiry: 6 hrs)
            user            // all of users data except password (see users.txt for more info)
        }
